\maketheorem{Stokes' theorem}{thm:stokesTheoremGeometricAlgebra:1740}{
\textit{Stokes' theorem} relates the dot product
of a
\(k\) volume element \(d^k \Bx \)
with the wedge product ``curl'' of an s-blade \( F, s < k \) as follows
\begin{equation*}
\int_V d^k \Bx \cdot (\boldpartial \wedge F) = \int_{\partial V} d^{k-1} \Bx \cdot F.
\end{equation*}
} % theorem
