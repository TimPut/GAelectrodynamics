\maketheorem{Reciprocal frame vectors}{thm:curvilinearGradient:1}{
Given a curvilinear basis with elements \( \Bx_i = \PDi{u_i}{\Bx} \), the reciprocal frame vectors are given by
\begin{dmath*}
\Bx^i = \spacegrad u_i.
\end{dmath*}
} % theorem
