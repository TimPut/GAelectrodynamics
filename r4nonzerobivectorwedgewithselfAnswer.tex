\makeanswer{problem:gradeselection:r4nonzerobivectorwedgewithself}{
\begin{dmath}\label{eqn:r4nonzerobivectorwedgewithself:20}
B \wedge B
=
\gpgrade{ B B }{4}
=
\gpgrade{
\lr{ \Be_1 \Be_2 + \Be_3 \Be_4 }
\lr{ \Be_1 \Be_2 + \Be_3 \Be_4 }
}{4}
=
\Be_3 \Be_4 \Be_1 \Be_2
+
\Be_1 \Be_2 \Be_3 \Be_4
=
2 \Be_1 \Be_2 \Be_3 \Be_4
\ne 0.
\end{dmath}
} % answer
