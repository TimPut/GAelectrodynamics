\maketheorem{Gauge invariance.}{thm:gaugeTransformation:60}{
The spacetime gradient of a grade \((0,3)\)-multivector \( \Psi \) may be added to a multivector potential
\begin{equation*}
A' = A + \stgrad \Psi,
\end{equation*}
without changing the field.
That is
\begin{equation*}
F
= \gpgrade{\conjstgrad A}{1,2}
= \gpgrade{\conjstgrad A'}{1,2}.
\end{equation*}
} % theorem
