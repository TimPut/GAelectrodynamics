\makedefinition{Adjoint.}{dfn:poyntingTheorem:1120}{
The \textit{adjoint} \( \overbar{A}(x) \) of a linear operator \( A(x) \) is defined implicitly by the scalar selection
\begin{equation*}
\gpgradezero{ y \overbar{A}(x) } =
\gpgradezero{ x A(y) }.
\end{equation*}
} % definition
