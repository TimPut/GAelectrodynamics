\maketheorem{Fundamental theorem for line integrals.}{thm:lineintegral:100}{
Given a
% piecewise-smooth
continuous curve
 \( C = \Bx(u) \)
parameterized by \( u \in [u_0, u_1] \), and multivector functions \( F(\Bx), G(\Bx) \) that are differentable over \( C \), then
\begin{equation*}
\int_C F d\Bx \lrboldpartial G
=
\evalbar{ F G }{\Delta u}
=
F(\Bx(u_1)) G(\Bx(u_1))
-
F(\Bx(u_0)) G(\Bx(u_0)).
\end{equation*}
} % theorem
