\maketheorem{Stokes' theorem for scalar functions.}{thm:lineintegral:180}{
Given a continuous curve \( C = \Bx(u) \) parameterized by parameter \( u \in [u_0, u_1] \), and a scalar function \(f(\Bx)\) differentable over \( C \), then
\begin{equation*}
\int_C d\Bx \cdot \spacegrad f
=
\int_C d\Bx \cdot \boldpartial f
= \evalbar{f}{\Delta u}.
\end{equation*}
} % theorem
